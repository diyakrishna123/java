import java.util.Scanner; class Product { int pcode; String pname; int price; // Constructor to initialize the product details Product(int code, String name, int cost) { this.pcode = code; this.pname = name; this.price = cost; } } public class Main { // Corrected the class name to Main public static void main(String[] args) { // Corrected the method signature Scanner input = new Scanner(System.in); // Prompt the user to enter the total number of products System.out.println("Enter the total number of products:"); int n = input.nextInt(); // Create an array to hold Product objects Product[] newpro = new Product[n]; // Loop to get details of each product for (int i = 0; i < n; i++) { System.out.println("Enter product code:"); int code = input.nextInt(); input.nextLine(); // Consume the newline character System.out.println("Enter the product name:"); String name = input.nextLine(); System.out.println("Enter the product price:"); int cost = input.nextInt(); // Create a new Product object and add it to the array newpro[i] = new Product(code, name, cost); } // Find the product with the lowest price int minPrice = newpro[0].price; Product minProduct = newpro[0]; for (int i = 1; i < n; i++) { if (newpro[i].price < minPrice) { minPrice = newpro[i].price; minProduct = newpro[i]; } } // Display the product with the lowest price System.out.println("Product with the lowest price: " + minProduct.pname + " with a price of: " + minProduct.price); } }
